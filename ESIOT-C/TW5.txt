//TW5 
//interface pushButton1 and ledPin1 for arduino UNO

constexpr uint8_t ledPin1 = A5;
constexpr uint8_t ledPin2 = A4;
constexpr uint8_t pushButton1 = 13;
constexpr uint8_t pushButton2 = 12;

bool buttonState1;
bool buttonState2;

void setup()
{
  Serial.begin(9600);
  buttonState1=false;
  buttonState2=false;
  pinMode(pushButton1, INPUT);
  pinMode(pushButton2, INPUT);
  pinMode(ledPin1, OUTPUT);
  pinMode(ledPin2, OUTPUT);
}

void loop()
{
  buttonState1 = digitalRead(pushButton1);
  buttonState2 = digitalRead(pushButton2);

  if(buttonState1 == LOW)
  {
    Serial.println("Button is pressed");
    digitalWrite(ledPin1, HIGH);
  }
  else
  {
    Serial.println("Button is released");
    digitalWrite(ledPin1, LOW);
  }

  if(buttonState2 == LOW)
  {
    Serial.println("Button is pressed");
    digitalWrite(ledPin2, HIGH);
  }
  else
  {
    Serial.println("Button is released");
    digitalWrite(ledPin2, LOW);
  }
}
