int light_pin = 5;

void setup() {
  // put your setup code here, to run once:
  pinMode(light_pin, INPUT);
  Serial.begin(9600);
}

void loop() {
  // put your main code here, to run repeatedly:
  int light_data = digitalRead(light_pin);
  if(light_data)
    Serial.println("No light");
  else
    Serial.println("light detected");
  delay(2000);
}
