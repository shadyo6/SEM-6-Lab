/*
Develop an Embedded ‘C’ program to control the relay through Arduino UNO.
Connection Details: Connect RM17 To RM9
*/

int relayPin = 8;
void setup() 
{
    Serial.begin(9600);
    pinMode(relayPin, OUTPUT);
    digitalWrite(relayPin, HIGH);
}

void loop() 
{
    digitalWrite(relayPin, LOW);
    delay(1000);
    digitalWrite(relayPin, HIGH);
    delay(1000);
}
