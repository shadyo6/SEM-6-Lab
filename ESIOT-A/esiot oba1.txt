#include <reg51.h>
void main(void){
	unsigned char data, count, i;
	data = P1;
	for(i = 0; i<8 ; i++){
		if((data >> i) & 0x01){
			count++;
		}
	}
	P2 = count;
}


#include <reg51.h>
void main(void){
	unsigned char num = 0x42;
	unsigned char upper = num >> 4;
	unsigned char lower = number & 0x0F;
	unsigned char temp = 0;
	temp = upper;
	upper = lower;
	lower = temp;
	num = (upper << 4) | lower;
}

#include <reg51.h>
void main(void){
	unsigned char numbers[4] = {0x76, 0x87, 0x98, 0x43};
	unsigned char converted[4];
	unsigned char i;
	for(i = 0; i<8 ; i++){
		unsigned char high = (numbers[i] & 0xF0) >> 4;
		unsigned char low = (numbers[i] & 0x0F);
		converted[i] = (high <= 9) ? high+'0' : high+'A'-10;
		converted[i] <<= 4;
		converted[i] = (low <= 9) ? low+'0' : low+'A'-10;
		P1 = converted;
	}
}
